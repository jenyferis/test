function switchSection(sectionId) { // Si ya estamos en esta sección, no hacer nada if (document.getElementById(sectionId).classList.contains('section-active')) { return; } // Desactivar todas las secciones actuales primero sections.forEach(section => { section.classList.remove('section-active'); }); // Actualizar los botones de navegación navButtons.forEach(btn => { btn.classList.remove('active'); if (btn.getAttribute('data-section') === sectionId) { btn.classList.add('active'); } }); // Cambiar fondo para la sección info if (sectionId === 'info-section') { body.classList.add('info-active'); } else { body.classList.remove('info-active'); } // Activar la nueva sección con un pequeño retraso para permitir que la transición de ocultamiento termine const targetSection = document.getElementById(sectionId); // Pequeño retraso para la animación fluida setTimeout(() => { targetSection.classList.add('section-active'); // Activar animaciones específicas para cada sección if (sectionId === 'gas-section' && animations.car) { animations.car.goToAndPlay(0, true); } else if (sectionId === 'info-section' && animations.bunny) { animations.bunny.goToAndPlay(0, true); } }, 100); // Aumentado ligeramente el tiempo de retraso } // Inicializar handlers de navegación navButtons.forEach(button => { button.addEventListener('click', function() { const sectionId = this.getAttribute('data-section'); switchSection(sectionId); }); });
